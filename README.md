# Requirement
    -   docker: 20.10.7
# Install
## clone the repository
    $ git clone git@github.com:FJCUFinaleTeam10/RMDP_api.git
## run the below command as root users
    $ cd RMDP_api
    $ pip install virtualenv
    $ virtualenv venv
    $ source venv/bin/activate
    $ docker-compose up -d
